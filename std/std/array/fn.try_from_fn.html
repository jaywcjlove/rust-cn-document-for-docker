let array: Result<[i8; 200], _> = std::array::try_from_fn(|i| i.try_into());

let array: Option<[_; 4]> = std::array::try_from_fn(|i| i.checked_add(100));

let array: Option<[_; 4]> = std::array::try_from_fn(|i| i.checked_sub(100));

特定于 Unix 的 fs::DirEntry 扩展方法。

Required Methods§

source

fn ino(&self) -> u64

返回所包含的 dirent 结构中的底层 d_ino 字段。

Examples
use std::fs;
use std::os::unix::fs::DirEntryExt;

if let Ok(entries) = fs::read_dir(".") {
    for entry in entries {
        if let Ok(entry) = entry {
            // 在此，`entry` 是 `DirEntry`。
            println!("{:?}: {}", entry.file_name(), entry.ino());
        }
    }
}

入门

启动 Erlang Shell

erl

编译代码

# 在 Erlang Shell 中编译
c(module).
# 在命令行中编译
erlc module.erl

运行代码

# 在 Erlang Shell 中运行
module:function().
# 从命令行运行
erl -noshell -s module function -s init stop

退出 Erlang Shell

q().

-module(module_name).
-export([function_name/arity, ...]).

function_name(Args) ->
    % Function body.
    Result.

导出函数

-export([function1/0, function2/1]).

% 单行注释

控制结构

条件语句

if
    Condition1 -> Expression1;
    Condition2 -> Expression2;
    true -> DefaultExpression
end.

case 表达式

case Expression of
    Pattern1 -> Expression1;
    Pattern2 -> Expression2;
    _ -> DefaultExpression
end.

函数定义

% 无参函数
my_function() ->
    ok.

% 有参函数
add(A, B) ->
    A + B.

列表操作

列表生成

% 生成 1 到 10 的列表
[ X || X <- lists:seq(1, 10)].

% 生成 1 到 10 中的偶数
[ X || X <- lists:seq(1, 10), X rem 2 == 0].

并发

启动进程

spawn(Module, Function, Args).

% 示例
Pid = spawn(fun() -> io:format("Hello from process~n") end).

发送消息

Pid ! Message.

% 示例
Pid ! {hello, self()}.

接收消息

receive
    Pattern1 -> Expression1;
    Pattern2 -> Expression2;
    after Timeout -> TimeoutExpression
end.

模式匹配

{ok, Value} = {ok, 42}.

示例：简单的服务器

-module(server).
-export([start/0, loop/0]).

start() ->
    spawn(fun loop/0).

loop() ->
    receive
        {echo, Msg} ->
            io:format("Echo: ~p~n", [Msg]),
            loop();
        stop ->
            io:format("Server stopping~n"),
            ok;
        _ ->
            io:format("Unknown message~n"),
            loop()
    end.

OTP 框架

定义 GenServer

-module(my_gen_server).
-behaviour(gen_server).

-export([start_link/0, init/1, handle_call/3, handle_cast/2, handle_info/2, terminate/2, code_change/3]).

start_link() ->
    gen_server:start_link({local, ?MODULE}, ?MODULE, [], []).

init([]) ->
    {ok, #state{}}.

handle_call(Request, From, State) ->
    {reply, Reply, State}.

handle_cast(Msg, State) ->
    {noreply, State}.

handle_info(Info, State) ->
    {noreply, State}.

terminate(Reason, State) ->
    ok.

code_change(OldVsn, State, Extra) ->
    {ok, State}.

使用 GenServer

gen_server:start_link({local, Name}, Module, Args, Options).
gen_server:call(ServerRef, Request).
gen_server:cast(ServerRef, Msg).

测试

编写 EUnit 测试

-module(module_name_tests).
-include_lib("eunit/include/eunit.hrl").

simple_test() ->
    ?assertEqual(Expected, Actual).

complex_test_() ->
    [
        {"Test case 1", ?_assertEqual(Expected1, Actual1)},
        {"Test case 2", ?_assertEqual(Expected2, Actual2)}
    ].

运行 EUnit 测试

# 在命令行中运行
erl -eval "eunit:test(module_name)" -s init stop
